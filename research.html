    i++;
    i++;

uchar DS18B20_ReadByte(void)                            //1Byte
{
    unsigned char i,j,dat;

    dat=0;
    for(i=1; i<=8; i++)
    {
        j = DS18B20_ReadBit();
        dat=(j<<7)|(dat>>1); 
    }

    return(dat);
}
void DS18B20_WriteByte(unsigned char dat)                //1Byte
{
    unsigned int i;
    unsigned char j, k;
    bit b;

    k = dat;

    for(j=1; j<=8; j++)
    {
        b = k & 0x01;
        k = k >> 1;
        if(b)                                             //1
        {
            DS_Data = 0;
            i++; i++;
            DS_Data = 1;
            i = 8;
            while(i>0) i--;
        }
        else
        {
            DS_Data = 0;                                   //0
            i = 8;
            while(i>0)    i--;
            DS_Data = 1;
            i++; i++;
        }

    }

    float t;

using System;
using System.Collections.Generic;
using System.Linq;
using System.Text;
using System.IO.Ports;

namespace IoT
{
    public class SerialControlHelper
    {
        SerialPort sp = null;                            //COM

        public bool SerialPortOpen(string Port)            //COM
        {
            sp = new SerialPort();                        //COM
            sp.PortName = Port;                            //
            sp.Parity = Parity.None;                    //
            sp.StopBits = StopBits.One;                    //1
            sp.BaudRate = 9600;                            //9600
            sp.DataBits = 8;                            //8
            sp.RtsEnable = true;    
            bool successOrNot = true;
            try
            {
                sp.Open();                                //
            }
            catch
            {
                successOrNot = false;                    //
            }
            return successOrNot;                        //
        }
        public Byte[] SerialData                        //SerialData
        {                                                //COM
            Get                                            //
            {
                string dataTemp = sp.ReadTo("X");        //X
                ASCIIEncoding ae = new ASCIIEncoding(); //ASCIIEncoding
                Byte[] dataByte = ae.GetBytes(dataTemp);    //ASCII
                return dataByte;                        //
            }
            Set                                            //
            {
                Byte[] dataByte = value;                //dataByte
                int dataLength = dataByte.Length;       //
                sp.Write(dataByte,0,dataLength);        //
            }
        }
        public bool SetTargetAddress(byte targetAddress)    //PC02F24
        {
            this.SerialData = new Byte[] {0x9d, 0x8c, 0x6b, 0x5a, 0xdd, 0x08, 0x01, targetAddress };    //
            bool successOrNot = false;
            if (sp.ReadByte() == 0x9d && sp.ReadByte() == 0x8c && sp.ReadByte() == 0x6b) successOrNot = true;    //9D 8C 6B
            return successOrNot;                        //
        }
        public void SendMessage(string messageText)        //
        {
            sp.Write(messageText);
        }
        public void SendAck()                            //Ack
        {
            sp.Write("AckX");
        }
        public bool GetAck()                            //Got
        {
            bool gotOrNot = false;
            if (sp.ReadTo("X") == "Got") gotOrNot = true;
            return gotOrNot;
        }
    }
}

        dat=(j<<7)|(dat>>1); 
    }

    return(dat);
}
void DS18B20_WriteByte(unsigned char dat)    
{
    unsigned int i;
    unsigned char j, k;
    bit b;

    k = dat;

    for(j=1; j<=8; j++)
    {
        b = k & 0x01;
        k = k >> 1;
        if(b)                     //write 1
        {
            DS_Data = 0;
            i++; i++;
            DS_Data = 1;
            i = 8;
            while(i>0) i--;
        }
        else
        {
            DS_Data = 0;               //write 0
            i = 8;
            while(i>0)    i--;
            DS_Data = 1;
            i++; i++;
        }

    }
}
void DS18B20_Change(void)    
{
    DS18B20_Reset();
    delayDSMs(5);
    DS18B20_WriteByte(0xcc);      // address all drivers on bus
    DS18B20_WriteByte(0x44);      //  initiates a single temperature conversion  
}
int DS18B20_GetTemp()            //get the temperature
{
    float t;
    unsigned char a, b;
    int temp;

    DS18B20_Reset();
    delayDSMs(5);
    DS18B20_WriteByte(0xcc);
    DS18B20_WriteByte(0xbe);
    a=DS18B20_ReadByte();
    b=DS18B20_ReadByte();

    temp = b;
    temp <<= 8;                     //two byte  compose a int variable
    temp = temp|a;
    t = temp*0.0625;
    temp = t*10+0.5;

    return temp;
}
